Car Details

#How Vehicle Table has been created
CREATE TABLE comp6000_06.Vehicle (
    vehicle_id INT UNIQUE AUTO_INCREMENT PRIMARY KEY,
    customer_id INT,
    registration_number VARCHAR(20) NOT NULL,
    vehicle_colour VARCHAR(50) NOT NULL,
    make VARCHAR(50) NOT NULL,
    year INT NOT NULL,
    fuel_type VARCHAR(50) NOT NULL,
    is_Deleted BOOLEAN NOT NULL DEFAULT FALSE,
    FOREIGN KEY (customer_id) REFERENCES Customer(customer_id)
);

Task 3: Display Upcoming Bookings

SELECT 
    b.booking_id,
    v.registration_number,
    v.make,
    GROUP_CONCAT(s.service_name SEPARATOR ', ') AS services,
    gl.location_name,
    b.booking_date,
    b.booking_start,
    b.booking_status
FROM 
    comp6000_06.Bookings AS b
JOIN 
    comp6000_06.Vehicle AS v ON b.vehicle_id = v.vehicle_id
JOIN 
    comp6000_06.BookingServices AS bs ON b.booking_id = bs.booking_id
JOIN 
    comp6000_06.Services AS s ON bs.service_id = s.service_id
JOIN 
    comp6000_06.Garage_Location AS gl ON b.location_id = gl.location_id
WHERE 
    b.booking_status = 'Active'
    AND b.customer_id = ? -- The customer that is logged on
GROUP BY 
    b.booking_id, v.registration_number, v.make, gl.location_name, b.booking_date, b.booking_start
ORDER BY 
    b.booking_date;
